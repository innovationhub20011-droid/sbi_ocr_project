CREATE TABLE PAN_CARD_DETAILS (
    ID                NUMBER GENERATED ALWAYS AS IDENTITY PRIMARY KEY,

    PAN_NUMBER        VARCHAR2(10 CHAR) NOT NULL,
    FULL_NAME         VARCHAR2(150 CHAR),
    FATHER_NAME       VARCHAR2(150 CHAR),
    DATE_OF_BIRTH     DATE,
    PAN_TYPE          VARCHAR2(20 CHAR),

    OCR_SOURCE        VARCHAR2(50 CHAR),
    IS_VERIFIED       CHAR(1) DEFAULT 'N' CHECK (IS_VERIFIED IN ('Y','N')),

    CREATED_BY        VARCHAR2(50 CHAR),
    CREATED_DATE      TIMESTAMP DEFAULT SYSTIMESTAMP,
    UPDATED_BY        VARCHAR2(10 CHAR),
    UPDATED_DATE      TIMESTAMP
);


ALTER TABLE PAN_CARD_DETAILS
ADD CONSTRAINT PAN_NUMBER_FORMAT_CHK
CHECK (
    REGEXP_LIKE(PAN_NUMBER, '^[A-Z]{5}[0-9]{4}[A-Z]$')
);



CREATE TABLE AADHAR_CARD_DETAILS (
    ID                 NUMBER GENERATED ALWAYS AS IDENTITY PRIMARY KEY,

    AADHAR_NUMBER      NUMBER(12,0) NOT NULL,
    FULL_NAME          VARCHAR2(150 CHAR),
    DATE_OF_BIRTH      DATE,
    GENDER             VARCHAR2(10 CHAR),
    ADDRESS            VARCHAR2(500 CHAR),

    OCR_SOURCE         VARCHAR2(50 CHAR),
    IS_VERIFIED        CHAR(1) DEFAULT 'N' CHECK (IS_VERIFIED IN ('Y','N')),

    CREATED_BY         VARCHAR2(50 CHAR),
    CREATED_DATE       TIMESTAMP DEFAULT SYSTIMESTAMP,
    UPDATED_BY         VARCHAR2(10 CHAR),
    UPDATED_DATE       TIMESTAMP
);
